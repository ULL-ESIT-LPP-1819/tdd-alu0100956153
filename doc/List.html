def size
	if !@head.nil? 

		n = @head
		size = 1
		while !n.next.nil? do
			size+=1
			n=n.next
		end
		return size
	end
	return 0
end

def sort
	if !@head.nil?
		n= @head
		array_valid = Array.new
		array_not_valid = Array.new
		array_res = Array.new(2)
		
		while !n.next.nil? do
			if n.value.sales < 6
				array_valid.push(n.value)
			else
				array_not_valid.push(n.value)
			end
			n=n.next
		end
		if n.value.sales < 6
			array_valid.push(n.value)
		else
			array_not_valid.push(n.value)
		end

		array_res[0] = array_valid
		array_res[1] = array_not_valid
		return array_res
	end
end
